// ── UTILS ──
const $ = id => document.getElementById(id);
const pad = n => String(n).padStart(2,'0');
const days   = ['Minggu','Senin','Selasa','Rabu','Kamis','Jumat','Sabtu'];
const months = ['Januari','Februari','Maret','April','Mei','Juni',
                'Juli','Agustus','September','Oktober','November','Desember'];

// ── CLOCK ──
function tick() {
  const now = new Date();
  const h = now.getHours(), m = now.getMinutes(), s = now.getSeconds();
  $('tH').textContent       = pad(h);
  $('tM').textContent       = pad(m);
  $('secN').textContent     = pad(s);
  $('secFill').style.width  = (s / 59 * 100) + '%';
  $('dateStr').textContent  = `${days[now.getDay()]} · ${now.getDate()} ${months[now.getMonth()]} ${now.getFullYear()}`;
  $('sbarLeft').textContent = `AOD · ${pad(h)}:${pad(m)}:${pad(s)}`;
}
tick();
setInterval(tick, 1000);

// ── BATTERY ──
function loadBatt() {
  if (!navigator.getBattery) {
    $('battPct').textContent = '';
    $('battFill').style.width = '60%';
    return;
  }
  navigator.getBattery().then(b => {
    const pct = Math.round(b.level * 100);
    $('battPct').textContent = pct + '%';
    $('battFill').style.width = pct + '%';
    const f = $('battFill');
    f.className = 'batt-fill ' + (b.charging ? 'bc' : pct > 50 ? 'bh' : pct > 20 ? 'bm' : 'bl');
    b.addEventListener('levelchange', loadBatt);
    b.addEventListener('chargingchange', loadBatt);
  });
}
loadBatt();

// ── MOON PHASE ──

  $('mIco').textContent  = p[i][0];
  $('mName').textContent = p[i][1];
})();

// ── WEATHER (Open-Meteo, no API key) ──
const wIco  = c => c===0?'☀️':c<=2?'🌤':c<=3?'☁️':c<=48?'🌫':c<=57?'🌧':c<=67?'🌨':c<=77?'❄️':c<=82?'🌦':'⛈';
const wDesc = c => c===0?'Cerah':c<=2?'Cerah Berawan':c<=3?'Mendung':c<=48?'Berkabut':c<=57?'Gerimis':c<=67?'Hujan':c<=82?'Hujan Lebat':'Badai';

function fetchW(lat, lon, city) {
  fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m,weather_code&hourly=temperature_2m,weather_code&timezone=auto&forecast_days=1`)
    .then(r => r.json()).then(d => {
      const t = Math.round(d.current.temperature_2m), c = d.current.weather_code;
      const ic = wIco(c), dc = wDesc(c);
      $('wT').textContent     = t; $('wIco').textContent   = ic; $('wD').textContent     = dc;
      $('wTBig').textContent  = t; $('wIcoBig').textContent= ic; $('wDBig').textContent  = dc;
      $('wLoc').textContent   = (city || 'LOKASI').toUpperCase();
      const row = $('wHourly'); row.innerHTML = '';
      const nh = new Date().getHours();
      for (let i = 1; i <= 4; i++) {
        const h = (nh+i)%24;
        const idx = d.hourly.time.findIndex(t => new Date(t).getHours()===h);
        const el = document.createElement('div'); el.className = 'wh';
        el.innerHTML = `<span class="wh-t">${pad(h)}:00</span><span class="wh-i">${idx>=0?wIco(d.hourly.weather_code[idx]):'?'}</span><span class="wh-v">${idx>=0?Math.round(d.hourly.temperature_2m[idx]):'--'}°</span>`;
        row.appendChild(el);
      }
    }).catch(() => { $('wD').textContent = 'Offline'; $('wDBig').textContent = 'Offline'; });
}

navigator.geolocation?.getCurrentPosition(
  p => fetch(`https://nominatim.openstreetmap.org/reverse?lat=${p.coords.latitude}&lon=${p.coords.longitude}&format=json`)
        .then(r=>r.json()).then(g=>fetchW(p.coords.latitude,p.coords.longitude,g.address?.city||g.address?.town||g.address?.village||''))
        .catch(()=>fetchW(p.coords.latitude,p.coords.longitude,'')),
  () => fetchW(-5.135399,119.412293,'Makassar')

  app.style.transform  = `translate(${cX}px,${cY}px)`;
}
function schedShift() { setTimeout(()=>{ shift(); schedShift(); }, (35+Math.random()*15)*1000); }
schedShift();

// V2 — COLOR ROTATION (setiap 15 menit)
const modes = ['','mode-b','mode-c']; let mIdx = 0;
function rotateColor() {
  document.body.classList.remove('mode-b','mode-c');
  mIdx = (mIdx+1) % 3;
  if (modes[mIdx]) document.body.classList.add(modes[mIdx]);
  $('biLabel').textContent = `PROTECT · AKTIF · PALETTE ${['A','B','C'][mIdx]}`;
}
setInterval(rotateColor, 15*60*1000);

// V3 — NIGHT DIMMING (00:00–06:00 → 25%)
function nightDim() {
  const h = new Date().getHours();
  document.body.style.filter = (h>=0 && h<6) ? 'brightness(0.25)' : '';
}
nightDim(); setInterval(nightDim, 60*1000);

// V4 — VISIBILITY SLEEP (tab tidak aktif → pause animasi)
document.addEventListener('visibilitychange', () => {
  const state = document.visibilityState === 'visible' ? 'running' : 'paused';
  document.querySelectorAll('*').forEach(el => el.style.animationPlayState = state);
  if (document.visibilityState === 'visible') { keepOn(); shift(); }
});

// V5 — STATIC DETECTION (idle >5 menit → force shift)
let lastAct = Date.now();
['click','touchstart','touchmove'].forEach(e => document.addEventListener(e, ()=>lastAct=Date.now()));
setInterval(()=>{ if ((Date.now()-lastAct)>5*60*1000){ cX=0; cY=0; shift(); lastAct=Date.now(); }}, 60*1000);

// V6 — PIXEL REFRESH (sweep putih tipis)
function pixelRefresh() {
  const ov = $('pixelRefresh'), sw = $('sweepBar');
  ov.style.display = 'block';
  ov.style.opacity = '0.07';
  setTimeout(()=>{
    ov.style.opacity = '0';
    setTimeout(()=>{ ov.style.display='none'; }, 600);
    sw.style.display = 'block'; sw.style.top = '0px';
    sw.style.transition = 'top 9s linear';
    setTimeout(()=>{ sw.style.top = window.innerHeight+'px'; }, 60);
    setTimeout(()=>{ sw.style.display='none'; sw.style.transition=''; sw.style.top='0px'; }, 9500);
  }, 3000);
  try { localStorage.setItem('aod_lr', Date.now()); } catch(e){}
  $('biLabel').textContent = 'PIXEL REFRESH · BERJALAN';
  setTimeout(()=>{ $('biLabel').textContent = `PROTECT · AKTIF · PALETTE ${['A','B','C'][mIdx]}`; }, 14000);
}

// Cek refresh mingguan tiap jam
setInterval(()=>{
  try {
    const lr = parseInt(localStorage.getItem('aod_lr')||'0');
    const h = new Date().getHours();
    if ((Date.now()-lr)>7*24*60*60*1000 && h>=1 && h<=4) pixelRefresh();
  } catch(e){}
}, 60*60*1000);

// Ketuk 5× cepat → manual refresh
let taps=0, tt=null;
document.addEventListener('click',()=>{
  taps++; clearTimeout(tt);
  tt = setTimeout(()=>taps=0, 1400);
  if (taps>=5){ taps=0; pixelRefresh(); }
});
